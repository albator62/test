Docker configuration for encrypted cloud file storage container

FROM ${image_from} MAINTAINER Josh Hogle josh.hogle@gmail.com

Versions

ENV CONTAINER_VERSION 1.1.0-2

Upgrade packages and install required packages for build

RUN \ apt-get update && \ apt-get -y dist-upgrade && \ apt-get -o Dpkg::Options::="--force-confold" --no-install-recommends -y install \ git python3-setuptools python3-appdirs python3-dateutil python3-requests python3-sqlalchemy \ python-fuse encfs python-software-properties software-properties-common && \ easy_install3 -U pip

Amazon Cloud Drive Client Install

RUN \ pip3 install certifi && \ pip3 install --upgrade git+https://github.com/yadayada/acd_cli.git

Google Drive FS Install

RUN \ add-apt-repository -y ppa:alessandro-strada/ppa && \ apt-get update && \ apt-get install google-drive-ocamlfuse && \ rm -rf /var/lib/apt/lists/* /tmp/* /var/tmp/*

Customizations

Add files to image

ADD files /

Configure image

RUN \ chmod 0755 /usr/local/sbin/*

Mount configuration folders

VOLUME [ "/etc/mount.d", "/var/lib/acd", "/var/lib/gdrive" ]
